## Installation

```php
composer require alexcaussades/bootstrap:dev-master
```

## Usage

```php
<?php

require 'vendor/autoload.php';
use Bootstrap\Alerts;
use Bootstrap\Bootstrap;
use Bootstrap\Form;

$boot = new Bootstrap();
$alerts = new Alerts();
?>
```

## HEAD Meta

````php

<?= $boot->getHeaderBootsrap(); ?>

````

## Container Col Row 

```php

/** Start */
<?=  $boot->startContainer();?>
<?=  $boot->startrow();?>
<?=  $boot->startcol();?>

/** End  */
<?=  $boot->endcol();?>
<?=  $boot->endrow();?>
<?=  $boot->endContainer();?>
```

## Alerts

```php
<?=  $alerts->setAlertsDimissing("danger", 'hello');?>
```
